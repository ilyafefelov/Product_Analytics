```python
import pandas as pd
from scipy import stats

# Load data
control = pd.read_csv('control_group.csv', sep=';')
test = pd.read_csv('test_group.csv', sep=';')

# Remove missing data day
control = control.dropna(subset=['# of Impressions'])

# Calculate CTR
control['CTR'] = control['# of Website Clicks'] / control['# of Impressions']
test['CTR'] = test['# of Website Clicks'] / test['# of Impressions']

# T-test for CTR
t_stat, p_value = stats.ttest_ind(
    control['CTR'].dropna(), 
    test['CTR'].dropna(), 
    equal_var=False
)

print(f"CTR t-test: t={t_stat:.4f}, p={p_value:.6f}")
# Output: CTR t-test: t=-3.9786, p=0.000339
```
